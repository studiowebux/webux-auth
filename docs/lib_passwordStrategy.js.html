"use strict";

/**
 * It compares a password to a custom regex
 * @param {String} regex
 * @param {String} password
 * @returns {Boolean}
 */
const checkPassword = (regex, password) => {
  if (password.match(regex)) {
    return true;
  }
  return false;
};

module.exports = { checkPassword };
